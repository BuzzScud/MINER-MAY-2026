"""
Simplified High-Edge Signal v2.0: decision-window-only entries, half-series trend,
Fib discount/premium/OTE, volume surge, ATR filter, phi-extension exits.
Called in 5-min-before to 5-min-after windows around the 20- and 50-minute marks each hour.
"""
from __future__ import annotations

from datetime import datetime
from types import SimpleNamespace
from typing import Any

from custom_math.statistics import statistics
from custom_math import golden_ratio
from custom_math import fibonacci


def _minute_of_hour(bar_data: Any) -> int | None:
    """Extract minute (0-59) from bar timestamp."""
    ts = getattr(bar_data, "timestamp", None)
    if ts is None:
        return None
    try:
        if hasattr(ts, "minute"):
            return ts.minute
        s = str(ts).replace("Z", "+00:00")
        dt = datetime.fromisoformat(s)
        return dt.minute
    except Exception:
        return None


def recent_swing_high_low(closes: list[float], lookback: int) -> tuple[float, float, int, int]:
    """
    Over the last lookback bars: min and max close and their indices.
    Returns (swing_low, swing_high, idx_low, idx_high). If len(closes) < 2 return (0, 0, 0, 0).
    """
    if len(closes) < 2:
        return (0.0, 0.0, 0, 0)
    use = closes[-lookback:] if lookback < len(closes) else closes
    if not use:
        return (0.0, 0.0, 0, 0)
    swing_low = min(use)
    swing_high = max(use)
    start = len(closes) - len(use)
    idx_low = start + use.index(swing_low)
    idx_high = start + use.index(swing_high)
    return (swing_low, swing_high, idx_low, idx_high)


def _atr_from_bars(bar_history: list[Any], current_bar: Any, period: int = 14) -> float:
    """Average True Range from high-low of bars. Fallback to 0 if no data."""
    bars = list(bar_history)
    if current_bar is not None:
        bars = (bars + [current_bar])[-(period + 1) :]
    if len(bars) < 2:
        return 0.0
    ranges: list[float] = []
    for b in bars:
        h = float(getattr(b, "high", 0) or 0)
        l = float(getattr(b, "low", 0) or 0)
        if h > 0 and l >= 0:
            ranges.append(h - l)
    if not ranges:
        return 0.0
    return sum(ranges[-period:]) / min(period, len(ranges))


def _symbol_eq(a: str, b: str) -> bool:
    """True if symbols refer to same asset (e.g. BTC/USD and BTCUSD)."""
    if a == b:
        return True
    return (a or "").replace("/", "").upper() == (b or "").replace("/", "").upper()


def _default_config() -> SimpleNamespace:
    return SimpleNamespace(
        USE_GOLDEN_FIB=True,
        VOLUME_SURGE_MULT=1.5,
        MIN_ATR_PULLBACK=0.5,
        FIB_ENTRY_TOLERANCE_PCT=2.0,
        BAR_HISTORY_MAX=120,
    )


def _serializable_value(v: Any) -> Any:
    """Convert value for JSON: replace NaN/Inf with None, keep int/float/bool/str/list/dict."""
    if v is None:
        return None
    if isinstance(v, float) and (v != v or v == float("inf") or v == float("-inf")):
        return None
    if isinstance(v, (int, bool, str)):
        return v
    if isinstance(v, float):
        return v
    if isinstance(v, (list, tuple)):
        return [_serializable_value(x) for x in v]
    if isinstance(v, dict):
        return {str(k): _serializable_value(x) for k, x in v.items()}
    return None


def get_signal_context(
    symbol: str,
    bar_data: Any,
    broker: Any,
    bar_history: list[Any] | None = None,
    *,
    config: Any = None,
) -> dict[str, Any] | None:
    """
    Compute the same formulas/values used by your_signal_function for buy/sell/hold.
    Returns a serializable dict for trade logging/learning; None if insufficient data.
    """
    if bar_history is None:
        bar_history = []
    cfg = config if config is not None else _default_config()
    tolerance_pct = getattr(cfg, "FIB_ENTRY_TOLERANCE_PCT", 2.0)

    closes = [float(getattr(b, "close", 0) or 0) for b in bar_history]
    volumes = [float(getattr(b, "volume", 0) or 0) for b in bar_history]
    current_close = float(getattr(bar_data, "close", 0) or 0)
    current_vol = float(getattr(bar_data, "volume", 0) or 0)
    if current_close and current_close == current_close:
        closes = (closes + [current_close])[-120:]
        volumes = (volumes + [current_vol])[-120:]

    if len(closes) < 30 or current_close <= 0:
        return None

    half = max(1, len(closes) // 2)
    mean_p = statistics(closes)["mean"]
    mean_diff = statistics(closes[half:])["mean"] - statistics(closes[:half])["mean"]
    mean_diff_pct = (mean_diff / mean_p * 100.0) if mean_p and mean_p > 0 else 0.0
    bullish_trend = mean_diff_pct > 0.8
    bearish_trend = mean_diff_pct < -0.8

    swing_low, swing_high, _, _ = recent_swing_high_low(closes, 30)
    if swing_high <= swing_low:
        return None

    in_discount = fibonacci.in_discount_zone(current_close, swing_high, swing_low)
    in_premium = fibonacci.in_premium_zone(current_close, swing_high, swing_low)
    in_ote = fibonacci.price_in_ote_zone(current_close, swing_high, swing_low, tolerance_pct)

    if len(volumes) >= 20:
        avg_vol = sum(volumes[-20:-1]) / 19
        volume_surge_mult = getattr(cfg, "VOLUME_SURGE_MULT", 1.5)
        volume_surge = current_vol > avg_vol * volume_surge_mult
    else:
        volume_surge = False

    atr = _atr_from_bars(bar_history, bar_data, 14)
    equilibrium = (swing_high + swing_low) / 2.0
    atr_dist = abs(current_close - equilibrium) / atr if atr > 0 else 0.0

    out = {
        "current_close": current_close,
        "mean_diff_pct": mean_diff_pct,
        "bullish_trend": bullish_trend,
        "bearish_trend": bearish_trend,
        "in_discount": in_discount,
        "in_premium": in_premium,
        "in_ote": in_ote,
        "volume_surge": volume_surge,
        "atr": atr,
        "atr_dist": atr_dist,
        "swing_low": swing_low,
        "swing_high": swing_high,
    }
    return {k: _serializable_value(v) for k, v in out.items()}


def your_signal_function(
    symbol: str,
    bar_data: Any,
    broker: Any,
    bar_history: list[Any] | None = None,
    *,
    config: Any = None,
) -> str:
    """
    Simplified High-Edge Signal v2.0: only fires in decision windows.
    Trend + Fib discount/premium/OTE + volume surge + ATR filter; phi-extension exits.
    Returns "buy" | "sell" | "hold".
    """
    if bar_history is None:
        bar_history = []
    cfg = config if config is not None else _default_config()
    focus_minutes = getattr(config, "FOCUS_MINUTES", (15, 20, 25, 45, 50, 55)) if config is not None else (15, 20, 25, 45, 50, 55)
    use_golden_fib = getattr(cfg, "USE_GOLDEN_FIB", True)
    volume_surge_mult = getattr(cfg, "VOLUME_SURGE_MULT", 1.5)
    tolerance_pct = getattr(cfg, "FIB_ENTRY_TOLERANCE_PCT", 2.0)

    minute = _minute_of_hour(bar_data)
    if minute not in focus_minutes:
        return "hold"

    closes = [float(getattr(b, "close", 0) or 0) for b in bar_history]
    volumes = [float(getattr(b, "volume", 0) or 0) for b in bar_history]
    current_close = float(getattr(bar_data, "close", 0) or 0)
    current_vol = float(getattr(bar_data, "volume", 0) or 0)

    if len(closes) < 30 or current_close <= 0:
        return "hold"

    closes = (closes + [current_close])[-120:]
    volumes = (volumes + [current_vol])[-120:]

    half = max(1, len(closes) // 2)
    mean_diff_raw = statistics(closes[half:])["mean"] - statistics(closes[:half])["mean"]
    mean_p = statistics(closes)["mean"]
    mean_diff_pct = (mean_diff_raw / mean_p * 100.0) if mean_p and mean_p > 0 else 0.0
    bullish_trend = mean_diff_pct > 0.8
    bearish_trend = mean_diff_pct < -0.8

    swing_low, swing_high, _, _ = recent_swing_high_low(closes, 30)
    if swing_high <= swing_low:
        return "hold"

    in_discount = fibonacci.in_discount_zone(current_close, swing_high, swing_low)
    in_premium = fibonacci.in_premium_zone(current_close, swing_high, swing_low)
    in_ote = fibonacci.price_in_ote_zone(current_close, swing_high, swing_low, tolerance_pct)

    if len(volumes) >= 20:
        avg_vol = sum(volumes[-20:-1]) / 19
        volume_surge = current_vol > avg_vol * volume_surge_mult
    else:
        volume_surge = False

    atr = _atr_from_bars(bar_history, bar_data, 14)
    equilibrium = (swing_high + swing_low) / 2.0
    atr_dist = abs(current_close - equilibrium) / atr if atr > 0 else 0.0

    if use_golden_fib:
        try:
            positions = broker.get_positions() if hasattr(broker, "get_positions") else []
            for pos in positions:
                psym = getattr(pos, "symbol", "") or getattr(pos, "symbol_id", "")
                if not _symbol_eq(psym, symbol):
                    continue
                avg_entry = float(getattr(pos, "avg_entry_price", 0) or 0)
                qty = float(getattr(pos, "qty", 0) or 0)
                if avg_entry <= 0 or qty == 0:
                    continue
                scale = atr if atr > 0 else (max(closes) - min(closes)) if closes else 0.0
                if scale <= 0:
                    scale = avg_entry * 0.01
                if qty > 0:
                    for level in golden_ratio.extension_levels(avg_entry, scale, "above"):
                        if current_close >= level:
                            return "sell"
                else:
                    for level in golden_ratio.extension_levels(avg_entry, scale, "below"):
                        if current_close <= level:
                            return "buy"
                break
        except Exception:
            pass

    if bullish_trend and (in_discount or in_ote) and volume_surge and atr > 0 and 0.5 <= atr_dist <= 4.0:
        return "buy"
    if bearish_trend and in_premium and volume_surge and atr > 0 and 0.5 <= atr_dist <= 4.0:
        return "sell"

    return "hold"
